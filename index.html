subir un repositorio

  1. // Iniciar un nuevo repositorio
    // Crear la carpeta oculta .git
    git init
  2. // Ver que archivos no han sido registrados
    git status
  3. // Agregar todos los archivos para que esté pendiente de los cambios
    git add .
  4. // Crear commit (fotografía del proyecto en ese momento)
    git commit -m "primer commit"
  5. // Muestra la lista de commit del mas reciente al más antigüo
    git log
